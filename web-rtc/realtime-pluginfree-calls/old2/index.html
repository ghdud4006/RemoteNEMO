How to setup voice-only call?

// www.RTCMultiConnection.org/latest.js

var connection = new RTCMultiConnection();
connection.session = {
    audio: true
};
connection.connect();
btnStartVoiceOnlyCall.onclick = function() {
    connection.open();
};
